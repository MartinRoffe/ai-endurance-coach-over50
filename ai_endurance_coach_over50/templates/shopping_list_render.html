{# Shared list builder — include after shopping_list_data.html and store link constants. #} function initShoppingList(opts) { const STORE = opts.store; const linkFn = opts.linkFn; const webFn = opts.webFn; const addLabel = opts.addLabel || "Add ▸"; const cycleLabel = opts.cycleLabel || ""; let activeCats = new Set(DEFAULT_SHOP_CATS); let weekMode = opts.weekMode === "recovery" ? "recovery" : "build"; const params = new URLSearchParams(location.search); const urlCats = params.get("cats"); if (urlCats) { const picked = urlCats.split(",").filter(function(c) { return SHOP_CATEGORY_IDS.indexOf(c) >= 0; }); if (picked.length) activeCats = new Set(picked); } else { const savedCats = localStorage.getItem("shopCategories"); if (savedCats) { try { const arr = JSON.parse(savedCats); if (Array.isArray(arr) && arr.length) { activeCats = new Set(arr.filter(function(c) { return SHOP_CATEGORY_IDS.indexOf(c) >= 0; })); } } catch (e) { /* ignore */ } } } const urlWeek = params.get("week"); if (urlWeek === "build" || urlWeek === "recovery") { weekMode = urlWeek; } else { const savedWeek = localStorage.getItem("shopWeekMode"); if (savedWeek === "build" || savedWeek === "recovery") weekMode = savedWeek; } const list = document.getElementById("list"); const cats = [...list.querySelectorAll(".cat h2")].map(function(h) { return h.parentElement; }); const itemRows = []; const tickState = new Map(); let weeklyTotal = 0, fullTotal = 0, total = 0, count = 0, weeklyCount = 0, added = 0; function catByName(n) { return cats.find(function(c) { return c.querySelector("h2").textContent.trim() === n; }); } function itemKey(name, note) { return name + "\0" + note; } function persistFilters() { localStorage.setItem("shopCategories", JSON.stringify([...activeCats])); localStorage.setItem("shopWeekMode", weekMode); const u = new URL(location.href); u.searchParams.delete("set"); u.searchParams.set("cats", [...activeCats].join(",")); u.searchParams.set("week", weekMode); history.replaceState(null, "", u); document.querySelectorAll("a.compare-link").forEach(function(a) { try { const link = new URL(a.getAttribute("href"), location.origin); link.searchParams.delete("set"); link.searchParams.set("cats", [...activeCats].join(",")); link.searchParams.set("week", weekMode); a.href = link.pathname + "?" + link.searchParams.toString(); } catch (e) { /* ignore */ } }); } function saveTicks() { tickState.clear(); list.querySelectorAll(".item").forEach(function(row) { const name = row.querySelector(".name").textContent; const note = row.querySelector(".note").textContent; if (row.classList.contains("done")) tickState.set(itemKey(name, note), true); }); } function clearItems() { saveTicks(); list.querySelectorAll(".item").forEach(function(el) { el.remove(); }); itemRows.length = 0; weeklyTotal = 0; fullTotal = 0; total = 0; count = 0; weeklyCount = 0; added = 0; } function renderTotals() { document.getElementById("tickedTotal").textContent = "£" + total.toFixed(2); const weeklyEl = document.getElementById("weeklyCoreTotal"); if (weeklyEl) weeklyEl.textContent = "£" + weeklyTotal.toFixed(2); document.getElementById("fullTotal").textContent = "£" + fullTotal.toFixed(2); const countEl = document.getElementById("count"); countEl.textContent = added + " ticked · " + count + " shown" + (weeklyCount ? " · " + weeklyCount + " weekly core" : ""); } function updateFilterUi() { document.querySelectorAll("[data-shop-cat]").forEach(function(btn) { if (btn.dataset.shopCat === "all") { btn.classList.toggle("active", activeCats.size === SHOP_CATEGORY_IDS.length); return; } btn.classList.toggle("active", activeCats.has(btn.dataset.shopCat)); }); document.querySelectorAll("[data-week-mode]").forEach(function(btn) { btn.classList.toggle("active", btn.dataset.weekMode === weekMode); }); const hint = document.getElementById("setHint"); if (!hint) return; const catLabels = [...activeCats].map(function(c) { return SHOP_CATEGORY_LABELS[c]; }); const catText = catLabels.length === SHOP_CATEGORY_IDS.length ? "all categories" : catLabels.join(", "); const weekLabel = weekMode === "recovery" ? "recovery week" : "build week"; const coreNote = activeCats.size <= 2 && activeCats.has("breakfast") && activeCats.has("lunch") ? " — ≈£26–30 core shop (batch chicken, oats, scotch eggs, lunches). Toggle Weekend ride only when you need rice cakes." : ""; hint.innerHTML = "Showing " + catText + " for the " + weekLabel + "" + (cycleLabel ? " · currently " + cycleLabel : "") + coreNote; } function renderList() { clearItems(); const GROUPS = shopGroupsFor(STORE, activeCats, weekMode); GROUPS.forEach(function(g) { const anchor = catByName(g.after); if (!anchor) return; let insertAfter = anchor; if (insertAfter.nextElementSibling && insertAfter.nextElementSibling.classList.contains("banner")) { insertAfter = insertAfter.nextElementSibling; } g.items.forEach(function(tuple) { const name = tuple[0], note = tuple[1], price = tuple[2], term = tuple[3], freq = tuple[4]; fullTotal += price; count++; if (freq === "weekly") { weeklyTotal += price; weeklyCount++; } const row = el("div", "item"); const chk = el("div", "check"); chk.textContent = ""; const meta = el("div", "meta"); meta.appendChild(el("div", "name", name)); meta.appendChild(el("div", "note", note)); const pr = el("div", "price", "£" + price.toFixed(2)); const a = el("a", "add" + (g.ext ? " ext" : ""), g.ext ? "Find ▸" : addLabel); a.href = g.ext ? webFn(term) : linkFn(term); a.target = "_blank"; a.rel = "noopener"; row.append(chk, meta, pr, a); const key = itemKey(name, note); if (tickState.get(key)) { row.classList.add("done"); chk.classList.add("on"); chk.textContent = "✓"; total += price; added++; } chk.onclick = function() { const on = row.classList.toggle("done"); chk.classList.toggle("on", on); chk.textContent = on ? "✓" : ""; if (on) { total += price; added++; } else { total -= price; added--; } renderTotals(); }; insertAfter.after(row); insertAfter = row; itemRows.push({ section: g.after, name: name, note: note, row: row }); }); }); renderTotals(); updateFilterUi(); updateZoneVisibility(); persistFilters(); } function updateZoneVisibility() { cats.forEach(function(catEl) { let el = catEl.nextElementSibling; let hasItems = false; while (el && !el.classList.contains("cat")) { if (el.classList.contains("item")) { hasItems = true; break; } el = el.nextElementSibling; } catEl.style.display = hasItems ? "" : "none"; }); } document.querySelectorAll("[data-shop-cat]").forEach(function(btn) { btn.addEventListener("click", function() { const id = btn.dataset.shopCat; if (id === "all") { activeCats = new Set(SHOP_CATEGORY_IDS); } else if (activeCats.has(id)) { if (activeCats.size === 1) return; activeCats.delete(id); } else { activeCats.add(id); } renderList(); }); }); document.querySelectorAll("[data-week-mode]").forEach(function(btn) { btn.addEventListener("click", function() { const mode = btn.dataset.weekMode === "recovery" ? "recovery" : "build"; if (mode === weekMode) return; weekMode = mode; renderList(); }); }); renderList(); return { getItemRows: function() { return itemRows; }, getActiveCategories: function() { return [...activeCats]; }, rebuild: renderList, }; }
